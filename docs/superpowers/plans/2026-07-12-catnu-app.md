# 貓奴修行 (catnu-app) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build and deploy 貓奴修行 — a single-file (`index.html`), localStorage-only, Cantonese, mobile-first cat-bonding tracker per `/Users/stephanieau/Downloads/CATNU_BUILD_SPEC.md`.

**Architecture:** One `index.html` holds all CSS (`<style>`) and all JS (`<script>`). The `<script>` is split into two zones, in this order:
1. **Pure-logic zone** — data constants + a `window.Catnu = {...}` object of pure functions (no DOM/localStorage access). Runs unconditionally at parse time.
2. **App zone** — `Catnu.Store` (localStorage read/write), DOM rendering, event wiring, all inside an `initApp()` function, invoked only via `window.addEventListener('DOMContentLoaded', initApp)`.

This split lets `tests/helpers/load-app.mjs` `vm`-eval `index.html`'s script against a stub `window`/`document` (whose `addEventListener` is a no-op) to unit-test the pure-logic zone with Node's built-in test runner — without adding any build step or shipping any test code behavior to production. Reference mockups already on disk (read during planning, safe to reuse as source): `~/Downloads/catlove-preview-retro.html` (retro tile CSS system — card/chip/quest/meter classes match spec §3 almost verbatim) and `~/Downloads/catlove-landing.html` (the landing page spec §2.5 says already exists, needs rebrand only).

**Tech Stack:** Vanilla HTML/CSS/JS (no framework, no bundler). Dev-only: Node ≥18 built-in `node --test` + `assert` for the pure-logic engine (not shipped). Deploy: static Vercel project `catnu-app`, pushed via a per-repo `github_push.py` (GitHub Git Data API), matching the pattern already used by `sales-trainer`, `daily-novel`, `venturenix-lab-seminar`.

**Insertion anchors** (created in Task 1, referenced by later tasks instead of line numbers since the file grows across tasks):
- CSS: `/* @CSS:TABS */` (every task appends its new component styles just above this line)
- JS pure-logic: `/* @LOGIC:DATA */` (constants), `/* @LOGIC:ENGINE */` (functions)
- JS app zone: `/* @APP:STORE */`, `/* @APP:INIT */` (functions called from `initApp`), `/* @APP:BOOT */` (the final `initApp()` wiring, always stays last)
- **HTML is entirely JS-rendered**, not static markup — `<div class="app" id="app"></div>` is the only body content (Task 1). Task 13 has `initApp()` render a fixed shell into `#app`: a header, a 5-tab bar, and five empty `<section id="tab-record">` / `tab-quests` / `tab-analysis` / `tab-almanac` / `tab-profile` containers. Every later tab task adds one `renderX(state)` function (at `/* @APP:INIT */`) that fills its own section via `innerHTML`/DOM calls, registered in a `RENDERERS = { record: renderRecord, ... }` map that a central `switchTab(name)` (also at `/* @APP:INIT */`) calls on tab click and on boot.

```js
// tests/helpers/load-app.mjs
import { readFileSync } from 'node:fs';
import { fileURLToPath } from 'node:url';
import vm from 'node:vm';
import path from 'node:path';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const INDEX_PATH = path.join(__dirname, '..', '..', 'index.html');

export function loadCatnu(storageSeed = {}) {
  const html = readFileSync(INDEX_PATH, 'utf8');
  const match = html.match(/<script>([\s\S]*?)<\/script>/);
  if (!match) throw new Error('no <script> block found in index.html');
  const scriptSrc = match[1];

  const store = new Map(Object.entries(storageSeed));
  const localStorage = {
    getItem: (k) => (store.has(k) ? store.get(k) : null),
    setItem: (k, v) => store.set(k, String(v)),
    removeItem: (k) => store.delete(k),
  };
  const stubWindow = { addEventListener() {}, localStorage };
  const stubDocument = {
    addEventListener() {},
    getElementById() { return null; },
    querySelector() { return null; },
    querySelectorAll() { return []; },
    createElement() { return { style: {}, classList: { add() {}, remove() {}, toggle() {} }, setAttribute() {}, appendChild() {} }; },
  };
  const context = {
    window: stubWindow,
    document: stubDocument,
    localStorage,
    console,
    Date,
    Math,
  };
  vm.createContext(context);
  vm.runInContext(scriptSrc, context, { filename: 'index.html#inline-script' });
  return context.window.Catnu;
}
```

```js
const DAY_MS = 86400000;
const DECAY = 0.85;

Catnu.entryPolarity = function entryPolarity(entry) {
  let pos = 0, neg = 0;
  for (const rid of (entry.reactions || [])) {
    const p = Catnu.REACTION_POLARITY[rid];
    if (p === 1) pos++; else if (p === -1) neg++;
  }
  if (pos > neg) return 'positive';
  if (neg > pos) return 'negative';
  return 'neutral';
};

function weightedWindowScore(logs, catId, nowTs, windowStartDaysAgo, windowEndDaysAgo) {
  let num = 0, den = 0, nonNeutralCount = 0;
  for (const entry of logs) {
    if (entry.catId !== catId) continue;
    const daysAgo = Math.floor((nowTs - entry.ts) / DAY_MS);
    if (daysAgo < windowStartDaysAgo || daysAgo > windowEndDaysAgo) continue;
    const polarity = Catnu.entryPolarity(entry);
    if (polarity === 'neutral') continue;
    nonNeutralCount++;
    const w = Math.pow(DECAY, daysAgo);
    den += w;
    if (polarity === 'positive') num += w;
  }
  if (nonNeutralCount < 5) return { score: null, insufficient: true };
  return { score: Math.round((100 * num) / den), insufficient: false };
}

Catnu.affectionScore = function affectionScore(logs, catId, nowTs) {
  const current = weightedWindowScore(logs, catId, nowTs, 0, 6);
  const previous = weightedWindowScore(logs, catId, nowTs, 7, 13);
  const weekChange = (!current.insufficient && !previous.insufficient)
    ? current.score - previous.score
    : null;
  return { score: current.score, insufficient: current.insufficient, weekChange };
};
```

**Design note (resolves ambiguity in spec §6):** `n` in every rule ("n≥5", "XX記錄得 n 次") is the count of entries where `actions` includes the action id — matching the mockup text "梳毛記錄得 3 次" and "抱佢 7 次有 5 次飛機耳". The **positive-finding** check (`P_a` vs `P_base`) uses the wider sample (action entries + same-cat entries logged within 30 min after each) because the spec explicitly names that window ("{action}後 30 分鐘"). The **negative-warning** check uses only the direct action entries' own co-occurring reactions (no 30-min window is named for it in the spec).

```js
Catnu.actionCorrelation = function actionCorrelation(logs, catId, actionId, nowTs) {
  const catLogs = logs.filter(e => e.catId === catId);
  const actionEntries = catLogs.filter(e => (e.actions || []).includes(actionId));
  const n = actionEntries.length;

  const THIRTY_MIN = 30 * 60000;
  const sample = [];
  for (const a of actionEntries) {
    if (!sample.includes(a)) sample.push(a);
    for (const e of catLogs) {
      if (e !== a && e.ts > a.ts && e.ts - a.ts <= THIRTY_MIN && !sample.includes(e)) {
        sample.push(e);
      }
    }
  }
  let samplePos = 0, sampleNeg = 0;
  for (const e of sample) {
    const p = Catnu.entryPolarity(e);
    if (p === 'positive') samplePos++; else if (p === 'negative') sampleNeg++;
  }
  const sampleNonNeutral = samplePos + sampleNeg;
  const P_a = sampleNonNeutral > 0 ? samplePos / sampleNonNeutral : null;

  let basePos = 0, baseNeg = 0;
  for (const e of catLogs) {
    const p = Catnu.entryPolarity(e);
    if (p === 'positive') basePos++; else if (p === 'negative') baseNeg++;
  }
  const baseNonNeutral = basePos + baseNeg;
  const P_base = baseNonNeutral > 0 ? basePos / baseNonNeutral : null;

  const NEG_IDS = ['flatears', 'walkaway', 'hiss', 'bite'];
  const negativeCount = actionEntries.filter(e =>
    (e.reactions || []).some(r => NEG_IDS.includes(r))
  ).length;
  const negativeRate = n > 0 ? negativeCount / n : 0;

  const insufficientData = n >= 1 && n < 5;
  const positiveFinding = n >= 5 && P_a !== null && P_base !== null && (P_a - P_base) >= 0.15;
  const negativeWarning = n >= 5 && negativeRate >= 0.6;

  return { n, P_a, P_base, negativeCount, negativeRate, insufficientData, positiveFinding, negativeWarning };
};
```

```js
      if (rule) text += ` ${rule.text}`;
      positives.push({ type: 'positive', actionId: action.id, text });

      notes.push({ type: 'insufficient', actionId: action.id, text: `${action.label} 記錄得 ${r.n} 次，未夠數據——今晚試多次？` });
    }

    for (const rule of breed.breedRules) {
      if (rule.when !== 'noRecentAction') continue;
      const last = logs

        .reduce((max, e) => Math.max(max, e.ts), -Infinity);
      const daysSince = last === -Infinity ? Infinity : (nowTs - last) / DAY_MS;
      if (daysSince >= rule.days) notes.push({ type: 'insufficient', actionId: rule.action, text: rule.text });
    }
  }
```

```js
function dateStrToDays(s) {
  const [y, m, d] = s.split('-').map(Number);
  return Math.floor(Date.UTC(y, m - 1, d) / DAY_MS);
}

Catnu.checkStreakBreak = function checkStreakBreak(streak, todayStr) {
  if (!streak.lastDoneDate) return { ...streak };
  const diff = dateStrToDays(todayStr) - dateStrToDays(streak.lastDoneDate);
  if (diff >= 2) return { ...streak, current: 0 };
  return { ...streak };
};

Catnu.completeQuestsForDay = function completeQuestsForDay(streak, todayStr) {
  if (streak.lastDoneDate === todayStr) return { ...streak };
  const current = streak.current + 1;
  return { current, best: Math.max(streak.best, current), lastDoneDate: todayStr };
};
```

```js
Catnu.checkNewMilestones = function checkNewMilestones(state, nowTs) {
  const unlocked = new Set(state.milestones.map(m => `${m.id}:${m.catId || ''}`));
  const found = [];

  for (const cat of state.cats) {
    for (const def of Catnu.MILESTONE_DEFS) {
      const key = `${def.id}:${cat.id}`;
      if (unlocked.has(key)) continue;
      if (def.check(state.logs, cat.id)) found.push({ id: def.id, catId: cat.id, unlockedAt: nowTs });
    }
    const score = Catnu.affectionScore(state.logs, cat.id, nowTs);
    if (!score.insufficient) {
      if (score.score >= 70 && !unlocked.has(`score-70:${cat.id}`)) found.push({ id: 'score-70', catId: cat.id, unlockedAt: nowTs });
      if (score.score >= 85 && !unlocked.has(`score-85:${cat.id}`)) found.push({ id: 'score-85', catId: cat.id, unlockedAt: nowTs });
    }
  }

  if (state.streak.best >= 7 && !unlocked.has('streak-7:')) found.push({ id: 'streak-7', catId: null, unlockedAt: nowTs });
  if (state.streak.best >= 30 && !unlocked.has('streak-30:')) found.push({ id: 'streak-30', catId: null, unlockedAt: nowTs });

  return found;
};
```

```js
function seedFromDate(dateStr) {
  let h = 0;
  for (let i = 0; i < dateStr.length; i++) h = (h * 31 + dateStr.charCodeAt(i)) >>> 0;
  return h;
}
function pick(arr, seed) { return arr[seed % arr.length]; }

function buildQuestForCategory(category, cat, logs, dateStr, seed, nowTs) {
  if (category === 'universal') {
    const text = pick(Catnu.UNIVERSAL_QUEST_POOL, seed);
    return { id: `${dateStr}-${cat.id}-u${seed}`, text, catId: cat.id, source: 'universal', done: false };
  }
  if (category === 'data') {
    let best = null, bestN = Infinity, fallback = null;
    for (const action of Catnu.ACTIONS) {
      const r = Catnu.actionCorrelation(logs, cat.id, action.id, nowTs);
      if (r.P_a !== null && (best === null || r.P_a > best.P_a)) best = { action, P_a: r.P_a };
      if (r.n < bestN) { bestN = r.n; fallback = action; }
    }
    if (best) return { id: `${dateStr}-${cat.id}-d`, text: `今日做多次 ${best.action.label}`, catId: cat.id, source: 'data', done: false };
    return { id: `${dateStr}-${cat.id}-d`, text: `試下 ${fallback.label}，收集數據`, catId: cat.id, source: 'data', done: false };
  }
  const breed = Catnu.BREEDS.find(b => b.id === cat.breedId);
  const rule = breed && breed.breedRules.find(r => r.when === 'noRecentAction');
  if (rule) return { id: `${dateStr}-${cat.id}-b`, text: rule.text, catId: cat.id, source: 'breed', done: false };
  if (breed && breed.mixedBreed) {
    const text = pick(breed.approachGuide, seed);
    return { id: `${dateStr}-${cat.id}-b`, text, catId: cat.id, source: 'breed', done: false };
  }
  const text = pick(Catnu.UNIVERSAL_QUEST_POOL, seed + 3);
  return { id: `${dateStr}-${cat.id}-b`, text, catId: cat.id, source: 'universal', done: false };
}

Catnu.generateDailyQuests = function generateDailyQuests(dateStr, cats, logs, nowTs) {
  const seed = seedFromDate(dateStr);
  const CATEGORIES = ['universal', 'data', 'breed'];
  const quests = [];
  for (let slot = 0; slot < 3; slot++) {
    const cat = cats[(seed + slot) % cats.length];
    const category = CATEGORIES[(seed + slot * 7) % CATEGORIES.length];
    quests.push(buildQuestForCategory(category, cat, logs, dateStr, seed + slot, nowTs));
  }
  return quests;
};
```

```js
function groomingNeedFor(breedId) {
  return Catnu.GROOMING_NEED[breedId] != null ? Catnu.GROOMING_NEED[breedId] : 35;
}

function quizReason(breed, desired) {
  if (desired.clingy >= 60 && breed.traits && breed.traits.clingy >= desired.clingy - 10) {
    return `${breed.name}黐人度高，啱你想要成日跟前跟後嘅陪伴感。`;
  }
  if (desired.active >= 60 && breed.traits && breed.traits.active >= 60) {
    return `${breed.name}活躍愛玩，配合到你屋企夠空間俾佢郁嘅生活模式。`;
  }
  if (desired.clingy <= 40 && breed.traits && breed.traits.clingy <= 45) {
    return `${breed.name}性格獨立，唔會日日纏住你，啱你想要嘅自由度。`;
  }
  return `${breed.name}整體性格特徵同你嘅生活模式最夾。`;
}

Catnu.scoreAllBreeds = function scoreAllBreeds(answers, opts) {
  opts = opts || {};
  const desired = { clingy: 50, active: 50, holdOk: 50, vocal: 50 };
  let groomTolerance = 55, beginner = false, budgetLow = false;

  Catnu.QUIZ_QUESTIONS.forEach((q, i) => {
    const opt = q.options[answers[i]];
    if (!opt) return;
    for (const key of Object.keys(opt.delta || {})) {
      desired[key] = Math.max(0, Math.min(100, desired[key] + opt.delta[key]));
    }
    if (opt.groomTolerance != null) groomTolerance = opt.groomTolerance;
    if (opt.beginner) beginner = true;
    if (opt.budgetLow) budgetLow = true;
  });

  const scored = Catnu.BREEDS.map(breed => {
    if (breed.mixedBreed) {
      return { breed, score: 60 + (budgetLow ? 15 : 0), reason: '性格靠實際相處摸底，唔受品種刻板定型限制，加上養育成本通常較低。' };
    }
    const dims = ['clingy', 'active', 'holdOk', 'vocal'];
    let diff = 0;
    for (const d of dims) diff += Math.abs(breed.traits[d] - desired[d]);
    let score = 100 - diff / dims.length;

    score -= Math.abs(groomingNeedFor(breed.id) - groomTolerance) * 0.15;
    if (beginner && breed.traits.active > 70) score -= 10;
    if (beginner && groomingNeedFor(breed.id) > 70) score -= 8;

    if (opts.mode === 'addCat') {
      score += ((breed.multiCatFriendly - 50) / 50) * 15;
      if ((opts.existingCatNegativeRate || 0) > 0.4) {
        score += ((50 - breed.traits.active) / 50) * 10 + ((50 - breed.traits.clingy) / 50) * 5;
      }
    }

    return { breed, score, reason: quizReason(breed, desired) };
  });

  scored.sort((a, b) => b.score - a.score);
  return scored;
};

Catnu.matchQuiz = function matchQuiz(answers, opts) {
  return Catnu.scoreAllBreeds(answers, opts).slice(0, 3);
};
```

```js
    if (!raw) return defaultState();

      const parsed = JSON.parse(raw);
```

```js
function roundRect(ctx, x, y, w, h, r) {
  ctx.beginPath();
  ctx.moveTo(x + r, y);
  ctx.arcTo(x + w, y, x + w, y + h, r);
  ctx.arcTo(x + w, y + h, x, y + h, r);
  ctx.arcTo(x, y + h, x, y, r);
  ctx.arcTo(x, y, x + w, y, r);
  ctx.closePath();
}

function wrapCenteredText(ctx, text, cx, y, maxWidth, lineHeight) {
  let line = '', lines = [];
  for (const ch of text.split('')) {
    const test = line + ch;
    if (ctx.measureText(test).width > maxWidth && line) { lines.push(line); line = ch; }
    else line = test;
  }
  if (line) lines.push(line);
  lines.forEach((l, i) => ctx.fillText(l, cx, y + i * lineHeight));
}
```
